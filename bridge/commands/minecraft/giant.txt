# TITLE: 巨大化
# SUBTITLE: {ListenerName}
# CATEGORY: 変身ステータス
# SOUND: entity.ravager.roar
# PARTICLE: minecraft:crit
// 巨人の力が宿る：攻撃力・耐性・ジャンプ力が大幅アップ（60秒）
// ※MC 1.20.1 は scale attribute 非対応（1.20.5+）のため見た目の巨大化は不可。効果で代替
execute at @p run playsound entity.ravager.roar master @a ~ ~ ~ 1.6 0.6
effect give @p minecraft:strength 60 1 true
effect give @p minecraft:resistance 60 0 true
effect give @p minecraft:jump_boost 60 1 true
effect give @p minecraft:absorption 60 1 true
execute at @p run particle minecraft:crit ~ ~1 ~ 2 2 2 0.2 120 force
execute at @p run particle minecraft:angry_villager ~ ~2.5 ~ 1 1 1 0.05 20 force
